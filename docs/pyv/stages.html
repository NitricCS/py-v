155    def decImm(self, opcode, inst):
156        """Decodes the immediate from the instruction word.
157
158        Args:
159            opcode: Opcode of current instruction.
160            inst: Current instruction word.
161
162        Returns:
163            The decoded immediate.
164        """
165
166        # Save sign bit
167        sign = getBit(inst, 31)
168
169        sign_ext = 0
170
171        imm = 0
172        # Decode + sign-extend immediate
173        if opcode in isa.INST_I:
174            imm_11_0 = getBits(inst, 31, 20)
175            imm = imm_11_0
176            if sign:
177                sign_ext = 0xfffff<<12
178        
179        elif opcode in isa.INST_S:
180            imm_11_5 = getBits(inst, 31, 25)
181            imm_4_0 = getBits(inst, 11, 7)
182            imm = (imm_11_5<<5) | imm_4_0
183            if sign:
184                sign_ext = 0xfffff<<12
185        
186        elif opcode in isa.INST_B:
187            imm_12 = getBit(inst, 31)
188            imm_10_5 = getBits(inst, 30, 25)
189            imm_4_1 = getBits(inst, 11, 8)
190            imm_11 = getBits(inst, 7, 7)
191            imm =  (imm_12<<12) | (imm_11<<11) | (imm_10_5<<5) | (imm_4_1<<1)
192            if sign:
193                sign_ext = 0x7ffff<<13
194        
195        elif opcode in isa.INST_U:
196            imm_31_12 = getBits(inst, 31, 12)
197            imm = imm_31_12<<12
198        
199        elif opcode in isa.INST_J:
200            imm_20 = getBit(inst, 31)
201            imm_10_1 = getBits(inst, 30, 21)
202            imm_11 = getBits(inst, 20, 20)
203            imm_19_12 = getBits(inst, 19, 12)
204            imm = (imm_20<<20) | (imm_19_12<<12) | (imm_11<<11) | (imm_10_1<<1)
205            if sign:
206                sign_ext = 0x7ff<<21
207
208        return (sign_ext | imm)

306    def alu(self, opcode, rs1, rs2, imm, pc, f3, f7):
307        """Implements arithmetic-logic unit (ALU)
308
309        Args: TODO
310            opcode ([type]): [description]
311            rs1 ([type]): [description]
312            rs2 ([type]): [description]
313            imm ([type]): [description]
314            pc ([type]): [description]
315            f3 ([type]): [description]
316            f7 ([type]): [description]
317
318        Returns:
319            ALU result.
320        """
321
322        # Helpers
323        def _slt(val1, val2):
324            """ SLT[I] instruction
325
326            Args:
327                val1: Value of register rs1
328                val2: rs2 / Sign-extended immediate
329            
330            Returns:
331                1 if val1 < val2 (signed comparison)
332                0 otherwise
333            """
334
335            msb_r = getBit(val1, 31)
336            msb_i = getBit(val2, 31)
337
338            # Check if both operands are positive
339            if (msb_r==0) and (msb_i==0):
340                if val1 < val2:
341                    return 1
342                else:
343                    return 0
344            # val1 negative; val2 positive
345            elif (msb_r==1) and (msb_i==0):
346                return 1
347            # val1 positive, val2 negative
348            elif (msb_r==0) and (msb_i==1):
349                return 0
350            # both negative
351            else:
352                if val2 < val1:
353                    return 1
354                else:
355                    return 0
356
357        def _sltu(val1, val2):
358            """ SLT[I]U instruction
359
360            Args:
361                val1: Value of register rs1
362                val2: rs2 / Sign-extended immediate
363            
364            Returns:
365                1 if val1 < val2 (unsigned comparison)
366                0 otherwise
367            """
368
369            if val1 < val2:
370                return 1
371            else:
372                return 0
373
374        def _sll(val1, val2):
375            """ SLL[I] instruction
376
377            Args:
378                val1: Value of register rs1
379                val2: rs2 / Immediate
380            
381            Returns:
382                Logical left shift of val1 by val2 (5 bits)
383            """
384
385            return (MASK_32 & (val1<<(0x1f&val2))) # Mask so that bits above bit 31 turn to zero (for Python)
386
387        def _srl(val1, val2):
388            """ SRL[I] instruction
389
390            Args:
391                val1: Value of register rs1
392                val2: rs2 / Immediate
393            
394            Returns:
395                Logical right shift of val1 by val2 (5 bits)
396            """
397
398            return (MASK_32 & (val1>>(0x1f&val2))) # Mask so that bits above bit 31 turn to zero (for Python)
399
400        def _sra(val1, val2):
401            """ SRA[I] instruction
402
403            Args:
404                val1: Value of register rs1
405                val2: rs2 / Immediate
406            
407            Returns:
408                Arithmetic right shift of val1 by val2 (5 bits)
409            """
410
411            msb_r = getBit(val1, 31)
412            shamt = 0x1f & val2
413            rshift = (MASK_32 & (val1>>shamt)) # Mask so that bits above bit 31 turn to zero (for Python)
414            if msb_r==0:
415                return rshift 
416            else:
417                # Fill upper bits with 1s
418                return (MASK_32 & (rshift | (0xffffffff<<(XLEN-shamt))))
419
420        # ------------------
421        # ALU start
422        # ------------------
423
424        # Select operands
425        op1 = op2 = 0
426        # op1
427        if opcode==isa.OPCODES['AUIPC'] or opcode==isa.OPCODES['JAL'] or opcode==isa.OPCODES['BRANCH']:
428            op1 = pc
429        else:
430            op1 = rs1
431        # op2
432        if opcode!=isa.OPCODES['OP']:
433            op2 = imm
434        else:
435            op2 = rs2
436
437        # Perform ALU op
438        alu_res = 0
439        if opcode==isa.OPCODES['LUI']:
440            alu_res = op2
441        
442        elif opcode==isa.OPCODES['AUIPC'] or opcode==isa.OPCODES['JAL'] or opcode==isa.OPCODES['BRANCH']:
443            alu_res = op1 + op2
444        
445        elif opcode==isa.OPCODES['JALR']:
446            alu_res = 0xfffffffe & (op1 + op2) # TODO: Why do we need mask here?
447
448        elif opcode==isa.OPCODES['LOAD'] or opcode==isa.OPCODES['STORE']: # TODO: Could be merged with the upper elif
449            alu_res = op1 + op2
450        
451        elif opcode==isa.OPCODES['OP-IMM']:
452            if f3==0b000: # ADDI
453                alu_res = op1 + op2
454            elif f3==0b010: # SLTI
455                alu_res = _slt(op1, op2)
456            elif f3==0b011: # SLTIU
457                alu_res = _sltu(op1, op2)
458            elif f3==0b100: # XORI
459                alu_res = op1 ^ op2
460            elif f3==0b110: # ORI
461                alu_res = op1 | op2
462            elif f3==0b111: # ANDI
463                alu_res = op1 & op2
464            elif f3==0b001: # SLLI
465                if f7==0: # TODO: We could remove this check if IDStage catches f7!=0 case
466                    alu_res = _sll(op1, op2)
467            elif f3==0b101:
468                if f7==0: # SRLI
469                    alu_res = _srl(op1, op2)
470                elif f7==0b0100000: # SRAI
471                    alu_res = _sra(op1, op2)
472        
473        elif opcode==isa.OPCODES['OP']:
474            if f7==0:
475                if f3==0b000: # ADD
476                    alu_res = op1 + op2
477                elif f3==0b001: # SLL
478                    alu_res = _sll(op1, op2)
479                elif f3==0b010: # SLT
480                    alu_res = _slt(op1, op2)
481                elif f3==0b011: # SLTU
482                    alu_res = _sltu(op1, op2)
483                elif f3==0b100: # XOR
484                    alu_res = op1 ^ op2
485                elif f3==0b101: # SRL
486                    alu_res = _srl(op1, op2)
487                elif f3==0b110: # OR
488                    alu_res = op1 | op2
489                elif f3==0b111: # AND
490                    alu_res = op1 & op2
491
492            elif f7==0b0100000:
493                if f3==0b000: # SUB
494                    alu_res = op1 - op2
495                elif f3==0b101: # SRA
496                    alu_res = _sra(op1, op2)
497   
498        return MASK_32 & alu_res

502    def branch(self, f3, rs1, rs2) -> bool:
503        """Performs comparison of rs1 and rs2 using comp op given by f3.
504
505        Returns:
506            True if branch is taken.
507        """
508
509        # Branch less-than (BLT) logic
510        def _blt(rs1, rs2):
511           if msb_32(rs1)==msb_32(rs2):
512               return rs1<rs2
513           elif msb_32(rs1)==1:
514               return True
515           else:
516               return False
517
518        if f3==0:               # BEQ
519            return rs1==rs2
520        elif f3==1:             # BNE
521            return rs1!=rs2
522        elif f3==4:             # BLT
523            return _blt(rs1, rs2)
524        elif f3==5:             # BGE
525            return not _blt(rs1, rs2)
526        elif f3==6:             # BLTU
527            return rs1<rs2
528        elif f3==7:             # BGEU
529            return rs1>=rs2

573        load_val = 0
574        # Don't write by default
575        self.mem.we = False
576        self.mem.re = False

595            elif f3 == 1: # SH
596                self.mem.writeRequest(addr, mem_wdata, 2)
597            elif f3 == 2: # SW
598                self.mem.writeRequest(addr, mem_wdata, 4)
599            else:

601        # else:
602        #     raise Exception('ERROR (MEMStage, process): Invalid op {}'.format(op))
603
604        # TODO: Illegal address execption handling
605
