 97class IDStage(Module):
 98    """Instruction decode stage.
 99
100    Inputs:
101        IFID_i: Interface from IFStage

103    Outputs:
104        IDEX_o: Interface to EXStage
105    """
106
107    def __init__(self, regf: Regfile):
108        super().__init__()
109        self.regfile = regf
110
111        # Inputs
112        self.IFID_i = Input(IFID_t)
113
114        # Outputs
115        self.IDEX_o = Output(IDEX_t)

117    def process(self):
118        # Read inputs
119        val: IFID_t = self.IFID_i.read()
120        inst = val.inst
121        self.pc = val.pc
122
123        # Illegal instruction if bits 1:0 of inst != b11
124        if (inst & 0x3) != 0x3:
125            raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
126
127        # Determine opcode (inst[6:2])
128        opcode = getBits(inst, 6, 2)
129
130        # funct3, funct7
131        funct3 = getBits(inst, 14, 12)
132        funct7 = getBits(inst, 31, 25)
133
134        self.check_exception(opcode, funct3, funct7)
135
136        # Determine register indeces
137        rs1_idx = getBits(inst, 19, 15)
138        rs2_idx = getBits(inst, 24, 20)
139        rd_idx = getBits(inst, 11, 7)
140
141        # Read regfile
142        rs1 = self.regfile.read(rs1_idx)
143        rs2 = self.regfile.read(rs2_idx)
144
145        # Decode immediate
146        imm = self.decImm(opcode, inst)
147
148        # Determine register file write enable
149        we = opcode in isa.REG_OPS
150
151        # Determine what to write-back into regfile
152        wb_sel = self.wb_sel(opcode)
153
154        # Determine none/load/store
155        mem = self.mem_sel(opcode)
156
157        # Outputs
158        self.IDEX_o.write(IDEX_t(rs1, rs2, imm, self.pc, rd_idx, we, wb_sel, opcode, funct3, funct7, mem))

160    def mem_sel(self, opcode):
161        """Generates control signal for memory access.
162
163        Args:
164            opcode: Opcode of current instruction.
165
166        Returns:
167            A special value when the instruction is LOAD/STORE.
168            0 otherwise.
169        """
170        if opcode==isa.OPCODES['LOAD']:
171            return LOAD
172        elif opcode==isa.OPCODES['STORE']:
173            return STORE
174        else:
175            return 0

177    def wb_sel(self, opcode):
178        """Generates control signal for write-back.
179
180        Args:
181            opcode: Opcode of current instruction.
182
183        Returns:
184            * 1: JAL instruction
185            * 2: LOAD instruction
186            * 0: otherwise
187        """
188        if opcode == isa.OPCODES['JAL']:
189            return 1
190        elif opcode == isa.OPCODES['LOAD']:
191            return 2
192        else:
193            return 0

195    def decImm(self, opcode, inst):
196        """Decodes the immediate from the instruction word.
197
198        Args:
199            opcode: Opcode of current instruction.
200            inst: Current instruction word.
201
202        Returns:
203            The decoded immediate.
204        """
205
206        # Save sign bit
207        sign = getBit(inst, 31)
208
209        sign_ext = 0
210
211        imm = 0
212        # Decode + sign-extend immediate
213        if opcode in isa.INST_I:
214            imm_11_0 = getBits(inst, 31, 20)
215            imm = imm_11_0
216            if sign:
217                sign_ext = 0xfffff<<12
218
219        elif opcode in isa.INST_S:
220            imm_11_5 = getBits(inst, 31, 25)
221            imm_4_0 = getBits(inst, 11, 7)
222            imm = (imm_11_5<<5) | imm_4_0
223            if sign:
224                sign_ext = 0xfffff<<12
225
226        elif opcode in isa.INST_B:
227            imm_12 = getBit(inst, 31)
228            imm_10_5 = getBits(inst, 30, 25)
229            imm_4_1 = getBits(inst, 11, 8)
230            imm_11 = getBits(inst, 7, 7)
231            imm =  (imm_12<<12) | (imm_11<<11) | (imm_10_5<<5) | (imm_4_1<<1)
232            if sign:
233                sign_ext = 0x7ffff<<13
234
235        elif opcode in isa.INST_U:
236            imm_31_12 = getBits(inst, 31, 12)
237            imm = imm_31_12<<12
238
239        elif opcode in isa.INST_J:
240            imm_20 = getBit(inst, 31)
241            imm_10_1 = getBits(inst, 30, 21)
242            imm_11 = getBits(inst, 20, 20)
243            imm_19_12 = getBits(inst, 19, 12)
244            imm = (imm_20<<20) | (imm_19_12<<12) | (imm_11<<11) | (imm_10_1<<1)
245            if sign:
246                sign_ext = 0x7ff<<21
247
248        return (sign_ext | imm)

250    def check_exception(self, opcode, f3, f7):
251        """[summary]
252
253        Args:
254            opcode ([type]): [description]
255            f3 ([type]): [description]
256            f7 ([type]): [description]
257
258        Returns:
259            [type]: [description]
260        """
261        illinst = False
262
263        if opcode not in isa.OPCODES.values():
264            # Illegal instruction
265            raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X}: unknown opcode")
266
267        if opcode==isa.OPCODES['OP-IMM']:
268            if f3==0b001 and f7!=0: # SLLI
269                # Illegal instruction
270                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
271                illinst = True
272            elif f3==0b101 and not(f7==0 or f7==0b0100000): # SRLI, SRAI
273                # Illegal instruction
274                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
275                illinst = True
276
277        if opcode==isa.OPCODES['OP']:
278            if not (f7==0 or f7==0b0100000):
279                # Illegal instruction
280                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
281                illinst = True
282            elif f7==0b0100000 and not(f3==0b000 or f3==0b101):
283                # Illegal instruction
284                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
285                illinst = True
286
287        if opcode==isa.OPCODES['JALR']:
288            if f3 != 0:
289                # Illegal instruction
290                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
291                illinst = True
292
293        if opcode==isa.OPCODES['BRANCH']:
294            if f3 == 2 or f3 == 3:
295                # Illegal instruction
296                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
297                illinst = True
298
299        if opcode==isa.OPCODES['LOAD']:
300            if f3 == 3 or f3 == 6 or f3 == 7:
301                # Illegal instruction
302                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
303                illinst = True
304
305        if opcode==isa.OPCODES['STORE']:
306            if f3 > 2:
307                # Illegal instruction
308                raise Exception(f"IDStage: Illegal instruction @ PC = 0x{self.pc:08X} detected.")
309                illinst = True
310
311        # TODO: do something with illinst
312
313        # TODO: Return some exception type
314        return False

358        # Read inputs
359        val: IDEX_t = self.IDEX_i.read()
360        opcode = val.opcode
361        rs1 = val.rs1
362        rs2 = val.rs2

364        pc = val.pc
365        f3 = val.funct3
366        f7 = val.funct7
367

389    def alu(self, opcode, rs1, rs2, imm, pc, f3, f7):
390        """Implements arithmetic-logic unit (ALU)
391
392        Args: TODO
393            opcode ([type]): [description]
394            rs1 ([type]): [description]
395            rs2 ([type]): [description]
396            imm ([type]): [description]
397            pc ([type]): [description]
398            f3 ([type]): [description]
399            f7 ([type]): [description]
400
401        Returns:
402            ALU result.
403        """
404
405        # Helpers
406        def _slt(val1, val2):
407            """ SLT[I] instruction
408
409            Args:
410                val1: Value of register rs1
411                val2: rs2 / Sign-extended immediate
412
413            Returns:
414                1 if val1 < val2 (signed comparison)
415                0 otherwise
416            """
417
418            msb_r = getBit(val1, 31)
419            msb_i = getBit(val2, 31)
420
421            # Check if both operands are positive
422            if (msb_r==0) and (msb_i==0):
423                if val1 < val2:
424                    return 1
425                else:
426                    return 0
427            # val1 negative; val2 positive
428            elif (msb_r==1) and (msb_i==0):
429                return 1
430            # val1 positive, val2 negative
431            elif (msb_r==0) and (msb_i==1):
432                return 0
433            # both negative
434            else:
435                if val2 < val1:
436                    return 1
437                else:
438                    return 0
439
440        def _sltu(val1, val2):
441            """ SLT[I]U instruction
442
443            Args:
444                val1: Value of register rs1
445                val2: rs2 / Sign-extended immediate
446
447            Returns:
448                1 if val1 < val2 (unsigned comparison)
449                0 otherwise
450            """
451
452            if val1 < val2:
453                return 1
454            else:
455                return 0
456
457        def _sll(val1, val2):
458            """ SLL[I] instruction
459
460            Args:
461                val1: Value of register rs1
462                val2: rs2 / Immediate
463
464            Returns:
465                Logical left shift of val1 by val2 (5 bits)
466            """
467
468            return (MASK_32 & (val1<<(0x1f&val2))) # Mask so that bits above bit 31 turn to zero (for Python)
469
470        def _srl(val1, val2):
471            """ SRL[I] instruction
472
473            Args:
474                val1: Value of register rs1
475                val2: rs2 / Immediate
476
477            Returns:
478                Logical right shift of val1 by val2 (5 bits)
479            """
480
481            return (MASK_32 & (val1>>(0x1f&val2))) # Mask so that bits above bit 31 turn to zero (for Python)
482
483        def _sra(val1, val2):
484            """ SRA[I] instruction
485
486            Args:
487                val1: Value of register rs1
488                val2: rs2 / Immediate
489
490            Returns:
491                Arithmetic right shift of val1 by val2 (5 bits)
492            """
493
494            msb_r = getBit(val1, 31)
495            shamt = 0x1f & val2
496            rshift = (MASK_32 & (val1>>shamt)) # Mask so that bits above bit 31 turn to zero (for Python)
497            if msb_r==0:
498                return rshift 
499            else:
500                # Fill upper bits with 1s
501                return (MASK_32 & (rshift | (0xffffffff<<(XLEN-shamt))))
502
503        # ------------------
504        # ALU start
505        # ------------------
506
507        # Select operands
508        op1 = op2 = 0
509        # op1
510        if opcode==isa.OPCODES['AUIPC'] or opcode==isa.OPCODES['JAL'] or opcode==isa.OPCODES['BRANCH']:
511            op1 = pc
512        else:
513            op1 = rs1
514        # op2
515        if opcode!=isa.OPCODES['OP']:
516            op2 = imm
517        else:
518            op2 = rs2
519
520        # Perform ALU op
521        alu_res = 0
522        if opcode==isa.OPCODES['LUI']:
523            alu_res = op2
524
525        elif (opcode==isa.OPCODES['AUIPC']
526              or opcode==isa.OPCODES['JAL']
527              or opcode==isa.OPCODES['BRANCH']
528              or opcode==isa.OPCODES['LOAD']
529              or opcode==isa.OPCODES['STORE']):
530            alu_res = op1 + op2
531
532        elif opcode==isa.OPCODES['JALR']:
533            alu_res = 0xfffffffe & (op1 + op2)
534
535        elif opcode==isa.OPCODES['OP-IMM']:
536            if f3==0b000: # ADDI
537                alu_res = op1 + op2
538            elif f3==0b010: # SLTI
539                alu_res = _slt(op1, op2)
540            elif f3==0b011: # SLTIU
541                alu_res = _sltu(op1, op2)
542            elif f3==0b100: # XORI
543                alu_res = op1 ^ op2
544            elif f3==0b110: # ORI
545                alu_res = op1 | op2
546            elif f3==0b111: # ANDI
547                alu_res = op1 & op2
548            elif f3==0b001: # SLLI
549                if f7==0: # TODO: We could remove this check if IDStage catches f7!=0 case
550                    alu_res = _sll(op1, op2)
551            elif f3==0b101:
552                if f7==0: # SRLI
553                    alu_res = _srl(op1, op2)
554                elif f7==0b0100000: # SRAI
555                    alu_res = _sra(op1, op2)
556
557        elif opcode==isa.OPCODES['OP']:
558            if f7==0:
559                if f3==0b000: # ADD
560                    alu_res = op1 + op2
561                elif f3==0b001: # SLL
562                    alu_res = _sll(op1, op2)
563                elif f3==0b010: # SLT
564                    alu_res = _slt(op1, op2)
565                elif f3==0b011: # SLTU
566                    alu_res = _sltu(op1, op2)
567                elif f3==0b100: # XOR
568                    alu_res = op1 ^ op2
569                elif f3==0b101: # SRL
570                    alu_res = _srl(op1, op2)
571                elif f3==0b110: # OR
572                    alu_res = op1 | op2
573                elif f3==0b111: # AND
574                    alu_res = op1 & op2
575
576            elif f7==0b0100000:
577                if f3==0b000: # SUB
578                    alu_res = op1 - op2
579                elif f3==0b101: # SRA
580                    alu_res = _sra(op1, op2)
581
582        return MASK_32 & alu_res

584    def branch(self, f3, rs1, rs2) -> bool:
585        """Performs comparison of rs1 and rs2 using comp op given by f3.
586
587        Returns:
588            True if branch is taken.
589        """
590
591        # Branch less-than (BLT) logic
592        def _blt(rs1, rs2):
593            if msb_32(rs1)==msb_32(rs2):
594                return rs1<rs2
595            elif msb_32(rs1)==1:
596                return True
597            else:
598                return False
599
600        if f3==0:               # BEQ
601            return rs1==rs2
602        elif f3==1:             # BNE
603            return rs1!=rs2
604        elif f3==4:             # BLT
605            return _blt(rs1, rs2)
606        elif f3==5:             # BGE
607            return not _blt(rs1, rs2)
608        elif f3==6:             # BLTU
609            return rs1<rs2
610        elif f3==7:             # BGEU
611            return rs1>=rs2

613    def check_exception(self, take_branch, alu_res, pc):
614        # --- Branch/jump target misaligned ------
615        if take_branch:
616            if alu_res & 0x3 != 0:
617                raise Exception(f"Target instruction address misaligned exception at PC = 0x{pc:08X}")
